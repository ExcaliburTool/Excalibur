public class ByteCompareInt {
    public static void main(String[] args) {
        byte b = 1;
        int i = 1;
        System.out.println(b == i);
        System.out.println(b + i);
        byte b1 = -1;
        System.out.println(b1 == 128);
        System.out.println(b1 != 128);
        System.out.println(b != b1);
        System.out.println(i != 1);
    }
}
